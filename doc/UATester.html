def options_menu
        @options = {}
        optparser = OptionParser.new do |opt|
                opt.banner = print_banner
                opt.separator ""

                @options[:help] = ""
                opt.on("-h", "--help", "Print this help message") do |op|
                        exit
                end
        
                @options[:url] = "localhost"
                opt.on("-u", "--url", "URL target to scan. Default it's <localhost>") do |op|
                        @scan.start_scan @options[:url]
                end
                
                @options[:browser] = ""
                opt.on("-b", "--browser", "Set a browser template for HTTP header, @options are: [firefox, safari and chrome]") do |op|
                end
        end
        return optparser
end

def run
        begin
                case ARGV[0]
                when "-u", "--url"
                        @options[:url] = ARGV[1]
                        puts "[+] target: <#{@options[:url]}>"
                when "-b", "--browser"
                        puts "TODO"
                else
                        puts @optparser
                end
                @optparser.parse!
        rescue => e
                puts e.message.capitalize
                exit 1
        end
end
