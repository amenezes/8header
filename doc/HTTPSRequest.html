def get_request(uri)
        con = Net::HTTP.new uri.host, uri.port
        con.read_timeout = DEFAULT_TIMEOUT
        con.use_ssl = true
        con.verify_mode = OpenSSL::SSL::VERIFY_NONE
        req=Net::HTTP::Get.new uri.request_uri
end
